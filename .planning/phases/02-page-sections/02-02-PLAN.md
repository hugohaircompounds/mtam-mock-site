    <script>
      document.addEventListener('DOMContentLoaded', () => {
        gsap.registerPlugin(ScrollTrigger);

        // Hero: fires on page load, no ScrollTrigger (visible immediately)
        gsap.fromTo("#hero .hero-content",
          { opacity: 0, y: 32 },
          { opacity: 1, y: 0, duration: 1.0, ease: "power3.out", delay: 0.2 }
        );

        // Struggle items: stagger reveal
        gsap.fromTo(".struggle-item",
          { opacity: 0, y: 40 },
          {
            opacity: 1, y: 0, duration: 0.8, stagger: 0.15, ease: "power2.out",
            scrollTrigger: { trigger: "#struggle", start: "top 75%", toggleActions: "play none none none" }
          }
        );

        // Manifesto: single element reveal
        gsap.fromTo("#manifesto .manifesto-content",
          { opacity: 0, y: 24 },
          {
            opacity: 1, y: 0, duration: 1.0, ease: "power2.out",
            scrollTrigger: { trigger: "#manifesto", start: "top 80%", toggleActions: "play none none none" }
          }
        );

        // Topic cards: stagger reveal
        gsap.fromTo(".topic-card",
          { opacity: 0, y: 40 },
          {
            opacity: 1, y: 0, duration: 0.8, stagger: 0.15, ease: "power2.out",
            scrollTrigger: { trigger: "#topics", start: "top 75%", toggleActions: "play none none none" }
          }
        );

        // Educator cards: stagger reveal
        gsap.fromTo(".educator-card",
          { opacity: 0, y: 40 },
          {
            opacity: 1, y: 0, duration: 0.8, stagger: 0.15, ease: "power2.out",
            scrollTrigger: { trigger: "#educators", start: "top 75%", toggleActions: "play none none none" }
          }
        );

        // Founders: single element reveal
        gsap.fromTo("#founders .founders-content",
          { opacity: 0, y: 24 },
          {
            opacity: 1, y: 0, duration: 1.0, ease: "power2.out",
            scrollTrigger: { trigger: "#founders", start: "top 80%", toggleActions: "play none none none" }
          }
        );

        // Testimonial cards: stagger reveal
        gsap.fromTo(".testimonial-card",
          { opacity: 0, y: 40 },
          {
            opacity: 1, y: 0, duration: 0.8, stagger: 0.15, ease: "power2.out",
            scrollTrigger: { trigger: "#testimonials", start: "top 75%", toggleActions: "play none none none" }

        );

        // Pricing cards: stagger reveal
